126// for glm::ivec4 map
127struct IVec4Hash
128{
129 std::size_t operator()(const glm::ivec4 &v) const noexcept
130 {
131 std::size_t h = 0;
132 h ^= std::hash<int>{}(v.x) + 0x9e3779b9 + (h << 6) + (h >> 2);
133 h ^= std::hash<int>{}(v.y) + 0x9e3779b9 + (h << 6) + (h >> 2);
134 h ^= std::hash<int>{}(v.z) + 0x9e3779b9 + (h << 6) + (h >> 2);
135 h ^= std::hash<int>{}(v.w) + 0x9e3779b9 + (h << 6) + (h >> 2);
136 return h;
137 }

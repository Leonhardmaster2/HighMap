134inline float approx_hypot(float a, float b)
135{
136 a = std::abs(a);
137 b = std::abs(b);
138 if (a > b) std::swap(a, b);
139 return 0.414f * a + b;
140}

148inline float approx_rsqrt(float a)
149{
150 union
151 {
152 float f;
153 uint32_t i;
154 } conv = {.f = a};
155 conv.i = 0x5f3759df - (conv.i >> 1);
156 conv.f *= 1.5F - (a * 0.5F * conv.f * conv.f);
157 return conv.f;
158}

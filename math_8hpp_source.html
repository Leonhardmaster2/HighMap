131inline float approx_hypot(float a, float b)
132{
133 a = std::abs(a);
134 b = std::abs(b);
135 if (a > b) std::swap(a, b);
136 return 0.414f * a + b;

145inline float approx_rsqrt(float a)
146{
147 union
148 {
149 float f;
150 uint32_t i;
151 } conv = {.f = a};
152 conv.i = 0x5f3759df - (conv.i >> 1);
153 conv.f *= 1.5F - (a * 0.5F * conv.f * conv.f);
154 return conv.f;
155}

132inline float approx_hypot(float a, float b)
133{
134 a = std::abs(a);
135 b = std::abs(b);
136 if (a > b) std::swap(a, b);
137 return 0.414f * a + b;
138}

146inline float approx_rsqrt(float a)
147{
148 union
149 {
150 float f;
151 uint32_t i;
152 } conv = {.f = a};
153 conv.i = 0x5f3759df - (conv.i >> 1);
154 conv.f *= 1.5F - (a * 0.5F * conv.f * conv.f);
155 return conv.f;
156}

1044template <typename Fn> inline void for_each_cell(Array &a, Fn &&fn)
1045{
1046 const int w = a.shape.x;
1047 const int h = a.shape.y;
1048
1049 for (int j = 0; j < h; ++j)
1050 for (int i = 0; i < w; ++i)
1051 fn(i, j, a(i, j));
1052}

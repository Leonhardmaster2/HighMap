57inline float bilinear_interp(float f00,
58 float f10,
59 float f01,
60 float f11,
61 float u,
62 float v)
63{
64 float a10 = f10 - f00;
65 float a01 = f01 - f00;
66 float a11 = f11 - f10 - f01 + f00;
67 return f00 + a10 * u + a01 * v + a11 * u * v;
68}

70inline float cubic_interpolate(float p[4], float x)
71{
72 return p[1] + 0.5 * x *
73 (p[2] - p[0] +
74 x * (2.0 * p[0] - 5.0 * p[1] + 4.0 * p[2] - p[3] +
75 x * (3.0 * (p[1] - p[2]) + p[3] - p[0])));
76}

67inline float bilinear_interp(float f00,
68 float f10,
69 float f01,
70 float f11,
71 float u,
72 float v)
73{
74 float a10 = f10 - f00;
75 float a01 = f01 - f00;
76 float a11 = f11 - f10 - f01 + f00;
77 return f00 + a10 * u + a01 * v + a11 * u * v;
78}

80inline float cubic_interpolate(float p[4], float x)
81{
82 return p[1] + 0.5 * x *
83 (p[2] - p[0] +
84 x * (2.0 * p[0] - 5.0 * p[1] + 4.0 * p[2] - p[3] +
85 x * (3.0 * (p[1] - p[2]) + p[3] - p[0])));
86}
